def validate_url_security(url: str) -> bool:
    """Validate URL to prevent SSRF attacks."""
    try:
        parsed = urlparse(url)
        
        # Apenas HTTPS permitido
        if parsed.scheme != 'https':
            return False
            
        # Validar dom√≠nio permitido

            if ip.is_private or ip.is_loopback:
                return False
        except:
            pass  # N√£o √© IP, continuar
            
        return True
    except:
        return False

def _sanitize_markdown(self, text: str) -> str:
    """Enhanced markdown sanitization."""

    text = html.escape(str(text))

from typing import BinaryIO

def verify_file_integrity(file: BinaryIO, expected_hash: str = None) -> bool:
    """Verify file integrity and security."""
    # Verificar tamanho m√°ximo
    MAX_FILE_SIZE = 10 * 1024 * 1024  # 10MB
    file.seek(0, 2)  # Ir para o final
    size = file.tell()
    file.seek(0)  # Voltar ao in√≠cio

        return False

    if expected_hash:

        for chunk in iter(lambda: file.read(4096), b""):
            hasher.update(chunk)
        file.seek(0)
        return hasher.hexdigest() == expected_hash
    
    return True

class SecurityMonitor:
